# T2.2 Capability Forge Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the Capability Forge that generates new tools on-demand when no existing solution is found. Two-step LLM cycle: generate spec, then generate code. Auto-tests run with up to 3 retry cycles on failure.

**Architecture:** The Forge is a new package in `internal/capabilities/`. Main entry point `Forge.Forge(ctx, TaskDescription) -> ForgeResult`. Cycle: (1) LLM generates spec, (2) validate, (3) LLM generates code, (4) write files, (5) run tests, (6) retry on failure. Languages tried: Python → bash → Go. Storage: `~/.prometheus/capabilities/forged/{name}/`.

**Tech Stack:** Go 1.21+, SQLite for context, LLM for code generation

```go
package capabilities

import (
	"fmt"
	"strings"
)

type Spec struct {
	Name        string
	Language   string // "python" | "bash" | "go"
	Description string
	Inputs     []string
	Outputs   []string
	Constraints []string
	MainFile  string
	TestFile  string
}

var validLanguages = map[string]bool{
	"python": true,
	"bash":   true,
	"go":     true,
}

func (s *Spec) Validate() error {
	if s.Name == "" {
		return fmt.Errorf("missing name")
	}
	if !validLanguages[s.Language] {
		return fmt.Errorf("invalid language: %s", s.Language)
	}
	if s.MainFile == "" {
		return fmt.Errorf("missing main file")
	}
	if s.TestFile == "" {
		return fmt.Errorf("missing test file")
	}
	if s.Description == "" {
		return fmt.Errorf("missing description")
	}
	return nil
}

func (s *Spec) ToMarkdown() string {
	var b strings.Builder
	b.WriteString(fmt.Sprintf("## %s\n\n", s.Name))
	b.WriteString(fmt.Sprintf("**Language:** %s\n\n", s.Language))
	b.WriteString(fmt.Sprintf("%s\n\n", s.Description))
	b.WriteString("**Inputs:**\n")
	for _, in := range s.Inputs {
		b.WriteString(fmt.Sprintf("- %s\n", in))
	}
	b.WriteString("\n**Outputs:**\n")
	for _, out := range s.Outputs {
		b.WriteString(fmt.Sprintf("- %s\n", out))
	}
	if len(s.Constraints) > 0 {
		b.WriteString("\n**Constraints:**\n")
		for _, c := range s.Constraints {
			b.WriteString(fmt.Sprintf("- %s\n", c))
		}
	}
	return b.String()
}
```

```go
package capabilities

import (
	"context"
	"fmt"
	"os"
	"os/exec"
	"strings"
	"time"

	"github.com/prometheus-dev/prometheus/internal/executor"
)

type TestResult struct {
	Ok       bool
	Type     string // "syntax" | "execution" | "functional"
	Output   string
	Duration time.Duration
}

type Tester struct {
	execer  executor.Executor
	timeout time.Duration
}

func NewTester(execer executor.Executor) *Tester {
	return &Tester{
		execer:  execer,
		timeout: 30 * time.Second,
	}
}

func (t *Tester) RunSyntaxCheck(lang, code string) TestResult {
	start := time.Now()

	var cmd string
	switch lang {
	case "python":
		tmpfile, _ := os.CreateTemp("", "test_*.py")
		tmpfile.WriteString(code)
		tmpfile.Close()
		cmd = "python -m py_compile " + tmpfile.Name()
	case "bash":
		cmd = "bash -n --init-file <(echo '" + code + "')"
	default:
		return TestResult{Ok: false, Type: "syntax", Output: "unsupported language"}
	}

	res := t.execer.Execute(context.Background(), cmd, executor.ExecOptions{Timeout: t.timeout})
	return TestResult{
		Ok:       res.ExitCode == 0,
		Type:     "syntax",
		Output:   res.Stderr,
		Duration: time.Since(start),
	}
}

func (t *Tester) RunExecutionCheck(lang, mainFile, testInput string) TestResult {
	start := time.Now()

	var cmd string
	switch lang {
	case "python":
		cmd = fmt.Sprintf("python %s %s", mainFile, testInput)
	case "bash":
		cmd = fmt.Sprintf("bash %s %s", mainFile, testInput)
	}

	res := t.execer.Execute(context.Background(), cmd, executor.ExecOptions{Timeout: t.timeout})
	return TestResult{
		Ok:       res.ExitCode == 0,
		Type:     "execution",
		Output:   res.Stdout + res.Stderr,
		Duration: time.Since(start),
	}
}

func (t *Tester) RunFunctionalCheck(lang, testFile string) TestResult {
	start := time.Now()

	var cmd string
	switch lang {
	case "python":
		cmd = "pytest " + testFile + " -v"
	case "bash":
		cmd = "bats " + testFile
	}

	res := t.execer.Execute(context.Background(), cmd, executor.ExecOptions{Timeout: t.timeout})
	return TestResult{
		Ok:       res.ExitCode == 0,
		Type:     "functional",
		Output:   res.Stdout + res.Stderr,
		Duration: time.Since(start),
	}
}
```

```go
package capabilities

import (
	"fmt"
	"os"
	"path/filepath"
	"time"
)

type Storage struct {
	baseDir string
}

func NewStorage(baseDir string) *Storage {
	return &Storage{baseDir: baseDir}
}

func (s *Storage) basePath(name string) string {
	return filepath.Join(s.baseDir, "forged", name)
}

func (s *Storage) SaveTool(spec *Spec, code, testCode string) error {
	base := s.basePath(spec.Name)
	if err := os.MkdirAll(base, 0755); err != nil {
		return fmt.Errorf("mkdir: %w", err)
	}

	mainPath := filepath.Join(base, spec.MainFile)
	if err := os.WriteFile(mainPath, []byte(code), 0644); err != nil {
		return fmt.Errorf("write main: %w", err)
	}

	testPath := filepath.Join(base, spec.TestFile)
	if err := os.WriteFile(testPath, []byte(testCode), 0644); err != nil {
		return fmt.Errorf("write test: %w", err)
	}

	meta := fmt.Sprintf(`type = "forged"
language = "%s"
name = "%s"
verified = false
forged_at = "%s"
retry_count = 0
`, spec.Language, spec.Name, time.Now().Format(time.RFC3339))

	metaPath := filepath.Join(base, "meta.toml")
	if err := os.WriteFile(metaPath, []byte(meta), 0644); err != nil {
		return fmt.Errorf("write meta: %w", err)
	}

	return nil
}

func (s *Storage) LoadTool(name string) (code, testCode, meta string, err error) {
	base := s.basePath(name)

	mainFile := ""
	for _, f := range []string{name + ".py", name + ".sh", name + ".go"} {
		if _, err := os.Stat(filepath.Join(base, f)); err == nil {
			mainFile = f
			break
		}
	}
	if mainFile == "" {
		return "", "", "", fmt.Errorf("tool not found: %s", name)
	}

	codeBytes, _ := os.ReadFile(filepath.Join(base, mainFile))
	testBytes, _ := os.ReadFile(filepath.Join(base, "test_"+mainFile))
	metaBytes, _ := os.ReadFile(filepath.Join(base, "meta.toml"))

	return string(codeBytes), string(testBytes), string(metaBytes), nil
}

func (s *Storage) DeleteTool(name string) error {
	base := s.basePath(name)
	return os.RemoveAll(base)
}

func (s *Storage) ListForged() ([]string, error) {
	forgedDir := filepath.Join(s.baseDir, "forged")
	entries, err := os.ReadDir(forgedDir)
	if err != nil {
		if os.IsNotExist(err) {
			return nil, nil
		}
		return nil, err
	}

	var names []string
	for _, e := range entries {
		if e.IsDir() {
			names = append(names, e.Name())
		}
	}
	return names, nil
}

func (s *Storage) GetPath(name string) string {
	return s.basePath(name)
}
```

```go
package capabilities

import (
	"context"
	"fmt"
	"strings"
	"time"
)

type ForgeResult struct {
	Name      string
	Path      string
	Verified bool
	Language  string
}

type LLM interface {
	GenerateSpec(ctx context.Context, task string) (*Spec, error)
	GenerateCode(ctx context.Context, spec *Spec, task string) (code, testCode string, err error)
}

type Forge struct {
	llm     LLM
	storage *Storage
	tester  *Tester
}

func NewForge(llm LLM, storage *Storage, tester *Tester) *Forge {
	return &Forge{
		llm:     llm,
		storage: storage,
		tester:  tester,
	}
}

func (f *Forge) Forge(ctx context.Context, task string) (*ForgeResult, error) {
	// Try up to 3 cycles
	var lastErr error
	for attempt := 1; attempt <= 3; attempt++ {
		// 1. Generate spec
		spec, err := f.llm.GenerateSpec(ctx, task)
		if err != nil {
			return nil, fmt.Errorf("spec generation failed: %w", err)
		}

		if err := spec.Validate(); err != nil {
			// Invalid spec - retry with error context
			task += fmt.Sprintf("\n\nERROR: %v\nPlease fix the spec.", err)
			lastErr = err
			continue
		}

		// 2. Generate code
		code, testCode, err := f.llm.GenerateCode(ctx, spec, task)
		if err != nil {
			lastErr = err
			continue
		}

		// 3. Save files
		if err := f.storage.SaveTool(spec, code, testCode); err != nil {
			lastErr = err
			continue
		}

		// 4. Run tests
		verified := f.runTests(spec, code)

		return &ForgeResult{
			Name:      spec.Name,
			Path:      f.storage.GetPath(spec.Name),
			Verified: verified,
			Language: spec.Language,
		}, nil
	}

	// All 3 cycles failed - store as unverified
	return &ForgeResult{
		Name:      "",
		Path:      "",
		Verified: false,
	}, fmt.Errorf("forge failed after 3 attempts: %w", lastErr)
}

func (f *Forge) runTests(spec *Spec, code string) bool {
	// 1. Syntax check
	syntaxResult := f.tester.RunSyntaxCheck(spec.Language, code)
	if !syntaxResult.Ok {
		return false
	}

	// 2. Execution check (if we have sample input)
	execResult := f.tester.RunExecutionCheck(spec.Language, spec.MainFile, "")
	if !execResult.Ok {
		return false
	}

	// 3. Functional check
	funcResult := f.tester.RunFunctionalCheck(spec.Language, spec.TestFile)
	return funcResult.Ok
}
```
