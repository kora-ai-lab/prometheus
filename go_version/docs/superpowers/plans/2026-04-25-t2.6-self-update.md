```go
package update

import (
    "encoding/json"
    "fmt"
    "net/http"
    "strings"

    "github.com/blang/semver"
)

type GitHubRelease struct {
    TagName string `json:"tag_name"`
}

func CheckForUpdate(owner, repo string) (hasUpdate bool, latestVersion string, err error) {
    url := fmt.Sprintf("https://api.github.com/repos/%s/%s/releases/latest", owner, repo)
    
    resp, err := http.Get(url)
    if err != nil {
        return false, "", fmt.Errorf("cannot check for updates: %w", err)
    }
    defer resp.Body.Close()

    if resp.StatusCode != http.StatusOK {
        return false, "", fmt.Errorf("GitHub API returned status %d", resp.StatusCode)
    }

    var release GitHubRelease
    if err := json.NewDecoder(resp.Body).Decode(&release); err != nil {
        return false, "", fmt.Errorf("parse release: %w", err)
    }

    latestVersion = strings.TrimPrefix(release.TagName, "v")
    return true, latestVersion, nil
}

func CompareVersions(local, remote string) (int, error) {
    localV, err := semver.Make(local)
    if err != nil {
        return 0, fmt.Errorf("invalid local version: %w", err)
    }
    remoteV, err := semver.Make(remote)
    if err != nil {
        return 0, fmt.Errorf("invalid remote version: %w", err)
    }
    return localV.Compare(remoteV), nil
}
```

```go
package update

import (
    "crypto/sha256"
    "fmt"
    "io"
    "net/http"
    "os"
    "path/filepath"
    "strings"
)

func DownloadAndVerify(binaryURL, expectedSHA256 string) (tmpPath string, err error) {
    // Clean expected hash (remove spaces, convert to lowercase)
    expectedSHA256 = strings.ToLower(strings.ReplaceAll(expectedSHA256, " ", ""))
    if len(expectedSHA256) != 64 {
        return "", fmt.Errorf("invalid SHA256 length: %d", len(expectedSHA256))
    }

    // Download binary
    resp, err := http.Get(binaryURL)
    if err != nil {
        return "", fmt.Errorf("download failed: %w", err)
    }
    defer resp.Body.Close()

    if resp.StatusCode != http.StatusOK {
        return "", fmt.Errorf("download returned status %d", resp.StatusCode)
    }

    // Create temp file
    tmpDir := os.TempDir()
    tmpPath = filepath.Join(tmpDir, "prometheus-update")
    
    f, err := os.Create(tmpPath)
    if err != nil {
        return "", fmt.Errorf("create temp file: %w", err)
    }
    defer f.Close()

    // Write to file and compute hash simultaneously
    h := sha256.New()
    writer := io.MultiWriter(f, h)

    _, err = io.Copy(writer, resp.Body)
    if err != nil {
        os.Remove(tmpPath)
        return "", fmt.Errorf("write failed: %w", err)
    }

    // Verify checksum
    actualHash := fmt.Sprintf("%x", h.Sum(nil))
    if actualHash != expectedSHA256 {
        os.Remove(tmpPath)
        return "", fmt.Errorf("checksum mismatch: expected %s, got %s", expectedSHA256, actualHash)
    }

    return tmpPath, nil
}
```

```go
package update

import (
    "fmt"
    "os"
    "os/exec"
    "path/filepath"
)

var (
    currentBinary string // Set by init or flag
)

func init() {
    currentBinary, _ = os.Executable()
}

func ApplyUpdate(tmpPath string) error {
    if _, err := os.Stat(tmpPath); err != nil {
        return fmt.Errorf("update binary not found: %w", err)
    }

    // Create backup of current binary
    backupPath := currentBinary + ".backup"
    if _, err := os.Stat(currentBinary); err == nil {
        if err := os.CopyFile(backupPath, currentBinary); err != nil {
            return fmt.Errorf("create backup: %w", err)
        }
    }

    // Replace binary
    if err := os.CopyFile(currentBinary, tmpPath); err != nil {
        // Try rollback from backup
        if _, err := os.Stat(backupPath); err == nil {
            os.Rename(backupPath, currentBinary)
        }
        return fmt.Errorf("replace binary: %w", err)
    }

    // Make executable
    if err := os.Chmod(currentBinary, 0755); err != nil {
        return fmt.Errorf("chmod: %w", err)
    }

    // Restart
    return restart()
}

func restart() error {
    args := os.Args
    cmd := exec.Command(currentBinary, args[1:]...)
    cmd.Stdin = os.Stdin
    cmd.Stdout = os.Stdout
    cmd.Stderr = os.Stderr
    return cmd.Start()
}

// os.CopyFile is a helper to copy a file
func os.CopyFile(dst, src string) error {
    from, err := os.Open(src)
    if err != nil {
        return err
    }
    defer from.Close()

    to, err := os.Create(dst)
    if err != nil {
        return err
    }
    defer to.Close()

    _, err = io.Copy(to, from)
    return err
}
```
