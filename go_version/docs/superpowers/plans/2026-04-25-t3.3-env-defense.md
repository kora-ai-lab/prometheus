```go
package security

import (
    "net"
    "strconv"
)

type PortResult struct {
    Port    int
    Service string
    Risk    string // "high", "medium", "low"
}

var portServices = map[int]string{
    21:   "ftp",
    22:   "ssh",
    23:   "telnet",
    25:   "smtp",
    53:   "dns",
    80:  "http",
    110: "pop3",
    143: "imap",
    443: "https",
    445: "smb",
    993: "imaps",
    995: "pop3s",
    1433: "mssql",
    1521: "oracle",
    3306: "mysql",
    3389: "rdp",
    5432: "postgres",
    6379: "redis",
    8080: "http-proxy",
    8443: "https-alt",
}

var highRiskPorts = []int{23, 135, 136, 137, 138, 139, 445, 3389, 5900}
var mediumRiskPorts = []int{21, 110, 143, 1433, 1521, 3306, 5432}

func ScanPort(host string, port int) (*PortResult, error) {
    addr := strconv.Itoa(port)
    conn, err := net.DialTimeout("tcp", host+":"+addr", timeout)
    if err != nil {
        return nil, err
    }
    conn.Close()

    service := portServices[port]
    if service == "" {
        service = "unknown"
    }

    risk := "low"
    for _, p := range highRiskPorts {
        if port == p {
            risk = "high"
            break
        }
    }
    if risk == "low" {
        for _, p := range mediumRiskPorts {
            if port == p {
                risk = "medium"
                break
            }
        }
    }

    return &PortResult{
        Port:    port,
        Service: service,
        Risk:    risk,
    }, nil
}

func ScanPorts(host string, ports []int) []*PortResult {
    var results []*PortResult
    for _, port := range ports {
        result, err := ScanPort(host, port)
        if err == nil {
            results = append(results, result)
        }
    }
    return results
}


const timeout = someDuration
```

```go
package security

import (
    "os"
    "path/filepath"
)

type PermissionFinding struct {
    Path     string
    Mode     os.FileMode
    Severity string
    Issue    string
}

func CheckPermissions(paths []string) []*PermissionFinding {
    var findings []*PermissionFinding
    
    for _, path := range paths {
        info, err := os.Lstat(path)
        if err != nil {
            continue
        }
        
        mode := info.Mode()
        
        // World-writable
        if mode&0002 != 0 {
            findings = append(findings, &PermissionFinding{
                Path:     path,
                Mode:     mode,
                Severity: "high",
                Issue:    "world-writable",
            })
        }
        
        // SUID/SGID scripts
        if mode&os.ModeSetuid != 0 || mode&os.ModeSetgid != 0 {
            findings = append(findings, &PermissionFinding{
                Path:     path,
                Mode:     mode,
                Severity: "high",
                Issue:    "SUID/SGID set",
            })
        }
    }
    
    return findings
}

func CheckSensitiveDirs(dirs []string) []*PermissionFinding {
    var findings []*PermissionFinding
    sensitive := []string{"/etc/passwd", "/etc/shadow", "/.ssh", "/.aws"}
    
    for _, dir := range sensitive {
        findings = append(findings, CheckPermissions([]string{dir})...)
    }
    
    return findings
}
```

```go
package dast

import (
    "crypto/tls"
    "net"
    "net/http"
    "time"
)

type Config struct {
    TargetURL     string
    Timeout       time.Duration
    FollowRedirects bool
}

type Result struct {
    URL              string
    StatusCode       int
    Headers          http.Header
    Server           string
    SecurityHeaders map[string]string
    TLSVersions     []string
    CipherSuites    []string
    Findings        []string
}

func NewDAST(cfg Config) *DAST {
    return &DAST{cfg: cfg}
}

type DAST struct {
    cfg Config
    client *http.Client
}

func (d *DAST) Scan() (*Result, error) {
    // Basic HTTP scan
    resp, err := d.client.Get(d.cfg.TargetURL)
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()
    
    result := &Result{
        URL:        d.cfg.TargetURL,
        StatusCode: resp.StatusCode,
        Headers:    resp.Header,
        Server:    resp.Header.Get("Server"),
    }
    
    // Check security headers
    result.SecurityHeaders = map[string]string{
        "Strict-Transport-Security": resp.Header.Get("Strict-Transport-Security"),
        "X-Frame-Options":          resp.Header.Get("X-Frame-Options"),
        "X-Content-Type-Options":  resp.Header.Get("X-Content-Type-Options"),
        "Content-Security-Policy": resp.Header.Get("Content-Security-Policy"),
    }
    
    return result, nil
}
```

```go
func (d *DAST) CheckTLS() (*TLSResult, error) {
    conn, err := tls.Dial("tcp", d.cfg.TargetURL, &tls.Config{InsecureSkipVerify: true})
    if err != nil {
        return nil, err
    }
    defer conn.Close()
    
    state := conn.ConnectionState()
    return &TLSResult{
        Version:     state.Version,
        CipherSuite: state.CipherSuiteName(),
    }, nil
}
```

```go
func (d *DAST) Fuzz(endpoints []string) []string {
    var found []string
    
    for _, endpoint := range endpoints {
        url := d.cfg.TargetURL + endpoint
        resp, err := d.client.Get(url)
        if err == nil && resp.StatusCode == 200 {
            found = append(found, endpoint)
        }
    }
    
    return found
}
```
