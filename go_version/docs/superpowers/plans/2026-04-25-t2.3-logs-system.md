```go
type Logger struct {
    mu           sync.Mutex
    file         *os.File
    currentDate  string        // "2025-01-15"
    sessionID   string        // UUID
    home        string        // ~/.prometheus
    writeCh     chan writeRequest
    done       chan struct{}
    pending    int
}

type writeRequest struct {
    level  string
    taskID string
    event map[string]any
}

func New(home string) (*Logger, error) {
    logDir := filepath.Join(home, "logs")
    if err := os.MkdirAll(logDir, 0o700); err != nil {
        return nil, err
    }

    today := time.Now().Format("2006-01-02")
    path := filepath.Join(logDir, today+".jsonl")
    file, err := os.OpenFile(path, os.O_CREATE|os.O_APPEND|os.O_WRONLY, 0o600)
    if err != nil {
        return nil, err
    }

    l := &Logger{
        file:        file,
        currentDate: today,
        home:       home,
        writeCh:    make(chan writeRequest, 512),
        done:       make(chan struct{}),
    }

    go l.writerLoop()
    return l, nil
}

func (l *Logger) writerLoop() {
    ticker := time.NewTicker(time.Second)
    defer ticker.Stop()

    for {
        select {
        case req := <-l.writeCh:
            l.writeSync(req.level, req.taskID, req.event)
            l.pending++
            if l.pending >= 100 {
                l.file.Sync()
                l.pending = 0
            }
        case <-ticker.C:
            if l.pending > 0 {
                l.file.Sync()
                l.pending = 0
            }
        case <-l.done:
            for len(l.writeCh) > 0 {
                req := <-l.writeCh
                l.writeSync(req.level, req.taskID, req.event)
            }
            l.file.Sync()
            return
        }
    }
}

func (l *Logger) writeSync(level, taskID string, event map[string]any) {
    // Check daily rotation
    today := time.Now().Format("2006-01-02")
    if today != l.currentDate {
        l.rotate(today)
    }

    // Redact secrets
    sanitized := redactMap(event)

    entry := LogEntry{
        Ts:      time.Now().UTC().Format(time.RFC3339Nano),
        Session: l.sessionID,
        TaskID:  taskID,
        Level:   level,
        Event:   sanitized,
    }

    data, _ := json.Marshal(entry)
    l.file.Write(data)
    l.file.Write([]byte("\n"))
}

func (l *Logger) rotate(today string) {
    l.file.Close()
    path := filepath.Join(l.home, "logs", today+".jsonl")
    file, _ := os.OpenFile(path, os.O_CREATE|os.O_APPEND|os.O_WRONLY, 0o600)
    l.file = file
    l.currentDate = today
}

func (l *Logger) Log(level, taskID string, event map[string]any) {
    select {
    case l.writeCh <- writeRequest{level, taskID, event}:
    default:
        // Channel full, write sync
        l.writeSync(level, taskID, event)
    }
}

func (l *Logger) Close() error {
    close(l.done)
    if l.file != nil {
        return l.file.Close()
    }
    return nil
}
```

```go
package logging

import (
    "bufio"
    "io"
    "os"
    "path/filepath"
    "strings"
    "time"

    "github.com/klauspost/compress/zstd"
)

type Compressor struct {
    logsDir string
    level  zstd.EncoderLevel
}

func NewCompressor(logsDir string) *Compressor {
    return &Compressor{
        logsDir: logsDir,
        level:  zstd.SpeedBestCompression,
    }
}

func (c *Compressor) CompressOldLogs(ctx context.Context) error {
    today := time.Now().Format("2006-01-02")
    entries, _ := os.ReadDir(c.logsDir)

    for _, entry := range entries {
        if ctx.Err() != nil {
            return ctx.Err()
        }

        name := entry.Name()
        if !strings.HasSuffix(name, ".jsonl") {
            continue
        }
        date := strings.TrimSuffix(name, ".jsonl")
        if date >= today {
            continue
        }

        srcPath := filepath.Join(c.logsDir, name)
        dstPath := srcPath + ".zst"

        if err := c.CompressFile(srcPath, dstPath); err != nil {
            continue
        }

        if c.VerifyIntegrity(dstPath) {
            os.Remove(srcPath)
        }
    }
    return nil
}

func (c *Compressor) CompressFile(src, dst string) error {
    in, err := os.Open(src)
    if err != nil {
        return err
    }
    defer in.Close()

    tmp := dst + ".tmp"
    out, err := os.Create(tmp)
    if err != nil {
        return err
    }

    enc, err := zstd.NewWriter(out, zstd.WithEncoderLevel(c.level))
    if err != nil {
        out.Close()
        os.Remove(tmp)
        return err
    }

    if _, err := io.Copy(enc, in); err != nil {
        enc.Close()
        out.Close()
        os.Remove(tmp)
        return err
    }
    enc.Close()
    out.Close()

    return os.Rename(tmp, dst)
}

func (c *Compressor) VerifyIntegrity(path string) bool {
    f, err := os.Open(path)
    if err != nil {
        return false
    }
    defer f.Close()

    dec, err := zstd.NewReader(f)
    if err != nil {
        return false
    }
    defer dec.Close()

    scanner := bufio.NewScanner(dec)
    count := 0
    for scanner.Scan() && count < 100 {
        var entry map[string]any
        if err := json.Unmarshal(scanner.Bytes(), &entry); err != nil {
            return false
        }
        count++
    }

    return count > 0
}
```

```go
package logging

import (
    "encoding/json"
    "fmt"
    "io"
    "os"
    "path/filepath"
    "strings"
    "time"

    "github.com/klauspost/compress/zstd"
    "github.com/kora-ai-lab/prometheus/internal/llm"
)

type Summarizer struct {
    logsDir    string
    summaryDir string
    provider  llm.Provider
}

type LogEntry struct {
    Ts      string         `json:"ts"`
    Session string        `json:"session"`
    TaskID  string        `json:"task_id,omitempty"`
    Level   string        `json:"level"`
    Event   map[string]any `json:"event"`
}

func NewSummarizer(logsDir, summaryDir string, provider llm.Provider) *Summarizer {
    return &Summarizer{
        logsDir:    logsDir,
        summaryDir: summaryDir,
        provider:  provider,
    }
}

func (s *Summarizer) SummarizeDay(ctx context.Context, date string) error {
    summaryPath := filepath.Join(s.summaryDir, date+".md")

    if _, err := os.Stat(summaryPath); err == nil {
        return nil // already exists
    }

    events, err := s.loadDayEvents(date)
    if err != nil {
        return fmt.Errorf("load events: %w", err)
    }
    if len(events) == 0 {
        return nil
    }

    digest := s.buildDigest(events)

    var summary string
    if s.provider != nil {
        summary, err = s.generateSummary(ctx, date, digest)
        if err != nil {
            return err
        }
    } else {
        summary = fallbackSummary(date, digest)
    }

    os.MkdirAll(s.summaryDir, 0700)
    return os.WriteFile(summaryPath, []byte(summary), 0600)
}

func (s *Summarizer) loadDayEvents(date string) ([]LogEntry, error) {
    var entries []LogEntry

    paths := []string{
        filepath.Join(s.logsDir, date+".jsonl"),
        filepath.Join(s.logsDir, date+".jsonl.zst"),
    }

    for _, path := range paths {
        if _, err := os.Stat(path); os.IsNotExist(err) {
            continue
        }

        f, err := os.Open(path)
        if err != nil {
            continue
        }
        defer f.Close()

        var reader io.Reader = f
        if strings.HasSuffix(path, ".zst") {
            dec, err := zstd.NewReader(f)
            if err != nil {
                continue
            }
            defer dec.Close()
            reader = dec
        }

        scanner := bufio.NewScanner(reader)
        for scanner.Scan() {
            var entry LogEntry
            if err := json.Unmarshal(scanner.Bytes(), &entry); err != nil {
                continue
            }
            entries = append(entries, entry)
        }

        break // found and loaded
    }

    return entries, nil
}

func (s *Summarizer) buildDigest(events []LogEntry) string {
    var sb strings.Builder

    stats := map[string]int{}
    var goals, files []string

    for _, e := range events {
        stats[e.Level]++

        if e.Level == "session" {
            if goal, ok := e.Event["goal"].(string); ok {
                goals = append(goals, goal)
            }
        }
        if e.Level == "create" {
            if path, ok := e.Event["path"].(string); ok {
                files = append(files, path)
            }
        }
    }

    sb.WriteString(fmt.Sprintf("STATS: %d events\n", len(events)))
    if len(goals) > 0 {
        sb.WriteString("Goals: " + strings.Join(goals, ", ") + "\n")
    }
    if len(files) > 0 {
        sb.WriteString("Files created: " + strings.Join(files[:5], ", ") + "\n")
    }

    return sb.String()
}

func (s *Summarizer) generateSummary(ctx context.Context, date, digest string) (string, error) {
    prompt := fmt.Sprintf(`Génère un résumé markdown concis de cette journée.

Données :
%s

Format :
# Journal %s

## En bref
[2-3 phrases]

## Accompli
[bullet list]

## Stats
%s`, digest, date, digest)

    resp, err := s.provider.Complete(ctx, []llm.Message{{Role: "user", Content: prompt}})
    if err != nil {
        return "", err
    }

    header := fmt.Sprintf("<!-- prometheus_summary_version:1 date:%s generated:%s -->\n",
        date, time.Now().Format(time.RFC3339))

    return header + resp.Content, nil
}

func fallbackSummary(date, digest string) string {
    return fmt.Sprintf(`<!-- prometheus_summary_version:1 date:%s generated:%s -->
# Journal %s

## En bref
Journée de travail enregistrée.

## Stats
%s
`, date, time.Now().Format(time.RFC3339), date, digest)
}
```
