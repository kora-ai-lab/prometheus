### Task 2: Add Restore Method

**Files:**
- Modify: `internal/context/manager.go`

- [ ] **Step 1: Add Restore method to manager.go**

Add after Snapshot method:

```go
func (m *Manager) Restore(data map[string]any) {
    if data == nil {
        return
    }
    if ws, ok := data["warm_summary"].(string); ok {
        m.warmSummary = ws
    }
    if hb, ok := data["hot_buffer"].([]any); ok {
        m.hotBuffer = make([]llm.Message, len(hb))
        for i, item := range hb {
            if msgMap, ok := item.(map[string]any); ok {
                if role, ok := msgMap["role"].(string); ok {
                    m.hotBuffer[i].Role = role
                }
                if content, ok := msgMap["content"].(string); ok {
                    m.hotBuffer[i].Content = content
                }
            }
        }
    }
}
```

Note: Need to handle JSON unmarshaling properly. Will be fixed in Step 2.

- [ ] **Step 2: Fix Restore to handle JSON unmarshal properly**

Replace the Restore method with proper unmarshaling:

```go
func (m *Manager) Restore(data map[string]any) error {
    if data == nil {
        return nil
    }
    if ws, ok := data["warm_summary"].(string); ok {
        m.warmSummary = ws
    }
    if hb, ok := data["hot_buffer"].(string); ok {
        var msgs []llm.Message
        if err := json.Unmarshal([]byte(hb), &msgs); err == nil {
            m.hotBuffer = msgs
        }
    }
    return nil
}

package context

import (
    "testing"

func TestManager_Snapshot(t *testing.T) {
    m := &Manager{
        hotBuffer: []llm.Message{
            {Role: "user", Content: "Hello"},
            {Role: "assistant", Content: "Hi there"},
        },
        warmSummary:   `{"goal":"test","done":[],"state":"active"}`,
        contextWindow: 4096,
    }

    snapshot := m.Snapshot()

    if snapshot == nil {
        t.Fatal("Snapshot returned nil")
    }

    if snapshot["warm_summary"] != m.warmSummary {
        t.Errorf("expected warm_summary %q, got %q", m.warmSummary, snapshot["warm_summary"])
    }
}
```
