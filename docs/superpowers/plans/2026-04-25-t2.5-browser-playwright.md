package browser

import (
    "context"
    "time"

    "github.com/playwright-community/playwright-go"
)

type PlaywrightClient struct {
    browser *playwright.Browser
    page   *playwright.Page
    capEngine *capabilities.Engine
}

func NewPlaywrightClient(home string, capEngine *capabilities.Engine) (*PlaywrightClient, error) {
    // Ensure playwright via Capability Engine
    if capEngine != nil {
        capEngine.Ensure(context.Background(), "playwright")
    }

    pw, err := playwright.Run()
    if err != nil {
        return nil, err
    }

    browser, err := pw.Chromium.Launch()
    if err != nil {
        return nil, err
    }

    page, err := browser.NewPage()
    if err != nil {
        browser.Close()
        return nil, err
    }

    return &PlaywrightClient{
        browser: browser,
        page:   page,
        capEngine: capEngine,
    }, nil
}

func (c *PlaywrightClient) Navigate(url string) error {
    return c.page.Goto(url)
}

func (c *PlaywrightClient) GetHTML() (string, error) {
    return c.page.Content()
}

func (c *PlaywrightClient) GetText() (string, error) {
    return c.page.TextContent("body")
}

func (c *PlaywrightClient) Click(selector string) error {
    return c.page.Click(selector)
}

func (c *PlaywrightClient) Fill(selector, value string) error {
    return c.page.Fill(selector, value)
}

func (c *PlaywrightClient) Submit(selector string) error {
    return c.page.Click(selector + " button[type=submit]")
}

func (c *PlaywrightClient) Screenshot() ([]byte, error) {
    return c.page.Screenshot()
}

func (c *PlaywrightClient) WaitForSelector(selector string, timeout time.Duration) error {
    ctx, cancel := context.WithTimeout(context.Background(), timeout)
    defer cancel()
    return c.page.WaitForSelector(selector, playwright.WaitForSelectorOptions{Timeout: float64(timeout.Milliseconds())})
}

func (c *PlaywrightClient) WaitForNavigation(timeout time.Duration) error {
    ctx, cancel := context.WithTimeout(context.Background(), timeout)
    defer cancel()
    return c.page.WaitForLoadState(playwright.LoadStateNetworkidle, playwright.WaitForLoadStateOptions{Timeout: float64(timeout.Milliseconds())})
}

func (c *PlaywrightClient) EvalJS(script string) (any, error) {
    return c.page.Evaluate(script)
}

func (c *PlaywrightClient) GetCookies() ([]*Cookie, error) {
    cookies, err := c.page.Context().Cookies()
    if err != nil {
        return nil, err
    }
    result := make([]*Cookie, len(cookies))
    for i, c := range cookies {
        result[i] = &Cookie{Name: c.Name, Value: c.Value}
    }
    return result, nil
}

func (c *PlaywrightClient) SetCookie(cookie *Cookie) error {
    _, err := c.page.Context().AddCookies([]playwright.Cookie{
        {Name: cookie.Name, Value: cookie.Value},
    })
    return err
}

func (c *PlaywrightClient) ScrollDown(pixels int) error {
    return c.page.Evaluate(fmt.Sprintf("window.scrollBy(0, %d)", pixels))
}

func (c *PlaywrightClient) Close() error {
    if c.browser != nil {
        return c.browser.Close()
    }
    return nil
}

func hasPlaywright() bool {
    // Check if playwright is available
    return true // simplified
}
```

package browser

import (
    "bytes"
    "context"
    "fmt"
    "os"
    "os/exec"
    "path/filepath"
    "time"

    "github.com/prometheus-dev/prometheus/internal/capabilities"
)

type ADB struct {
    device   string
    capEngine *capabilities.Engine
}

func NewADB(capEngine *capabilities.Engine) (*ADB, error) {
    // Ensure adb via Capability Engine
    if capEngine != nil {
        capEngine.Ensure(context.Background(), "adb")
    }

    // Get device ID
    cmd := exec.Command("adb", "devices")
    var out bytes.Buffer
    cmd.Stdout = &out
    cmd.Run()

    device := "default"
    return &ADB{device: device, capEngine: capEngine}, nil
}

func (a *ADB) Screenshot() (string, error) {
    tmpDir := os.TempDir()
    localPath := filepath.Join(tmpDir, fmt.Sprintf("adb_screenshot_%d.png", time.Now().Unix()))

    cmd := exec.Command("adb", "-s", a.device, "exec-out", "screencap", "-p")
    var out bytes.Buffer
    cmd.Stdout = &out
    if err := cmd.Run(); err != nil {
        return "", fmt.Errorf("screencap: %w", err)
    }

    if err := os.WriteFile(localPath, out.Bytes(), 0600); err != nil {
        return "", fmt.Errorf("write file: %w", err)
    }

    return localPath, nil
}

func (a *ADB) Tap(x, y int) error {
    cmd := exec.Command("adb", "-s", a.device, "shell", "input", "tap", fmt.Sprintf("%d", x), fmt.Sprintf("%d", y))
    return cmd.Run()
}

func (a *ADB) InputText(text string) error {
    cmd := exec.Command("adb", "-s", a.device, "shell", "input", "text", text)
    return cmd.Run()
}

func (a *ADB) Close() error {
    return nil
}

func hasADB() bool {
    cmd := exec.Command("adb", "version")
    return cmd.Run() == nil
}
```

package browser

import (
    "strings"
)

var captchaSelectors = []string{
    "captcha",
    "g-recaptcha",
    "h-captcha",
    "recaptcha",
    "data-sitekey",
    "cf-challenge",
    "challenge",
}

func DetectCAPTCHA(html string) bool {
    lower := strings.ToLower(html)
    for _, selector := range captchaSelectors {
        if strings.Contains(lower, selector) {
            return true
        }
    }
    return false
}
