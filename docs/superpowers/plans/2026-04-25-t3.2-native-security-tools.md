package sast

import (
    "context"
    "os"
    "path/filepath"
    "regexp"
    "strings"
    "time"
)

type Severity string

const (
    Critical Severity = "CRITICAL"
    High     Severity = "HIGH"
    Medium   Severity = "MEDIUM"
    Low      Severity = "LOW"
    Info     Severity = "INFO"
)

type Finding struct {
    RuleID  string
    File   string
    Line   int
    Code   string
    Severity Severity
}

type FindingSet struct {
    File     string
    Findings []*Finding
}

type Scanner struct {
    rules []*Rule
}

type Rule struct {
    ID      string
    Pattern *regexp.Regexp
}

// NewScanner creates a new SAST scanner
func NewScanner() *Scanner {
    return &Scanner{
        rules: loadAllRules(),
    }
}

func loadAllRules() []*Rule {
    var rules []*Rule
    rules = append(rules, getSecretsRules()...)
    rules = append(rules, getInjectionRules()...)
    rules = append(rules, getAuthRules()...)
    return rules
}

// ScanFile scans a single file
func (s *Scanner) ScanFile(ctx context.Context, path string) (*FindingSet, error) {
    content, err := os.ReadFile(path)
    if err != nil {
        return nil, err
    }

    set := &FindingSet{File: path}
    lines := strings.Split(string(content), "\n")

    for lineNum, line := range lines {
        for _, rule := range s.rules {
            if rule.Pattern.MatchString(line) {
                set.Findings = append(set.Findings, &Finding{
                    RuleID:   rule.ID,
                    File:    path,
                    Line:    lineNum + 1,
                    Code:    strings.TrimSpace(line),
                    Severity: High,
                })
            }
        }
    }

    return set, nil
}

// ScanDir scans all code files in a directory
func (s *Scanner) ScanDir(ctx context.Context, dir string) ([]*FindingSet, error) {
    var results []*FindingSet

    filepath.WalkDir(dir, func(path string, d os.DirEntry, err error) error {
        if err != nil { return err }
        if d.IsDir() {
            name := d.Name()
            if name == "node_modules" || name == ".git" || name == "__pycache__" ||
               name == "vendor" || name == ".venv" {
                return filepath.SkipDir
            }
            return nil
        }

        ext := strings.ToLower(filepath.Ext(path))
        if ext == ".py" || ext == ".js" || ext == ".ts" || ext == ".go" ||
           ext == ".sh" || ext == ".sql" {
            set, err := s.ScanFile(ctx, path)
            if err == nil && len(set.Findings) > 0 {
                results = append(results, set)
            }
        }
        return nil
    })

    return results, nil
}
```

func TestSecretsRules_Match(t *testing.T) {
    rules := getSecretsRules()
    
    tests := []struct {
        code    string
        wantID  string
    }{
        {`api_key = "sk-abcdef123456789"`, "SEC001"},
        {`password = "secret123"`, "SEC002"},
        {`jwt = "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9..."`, "SEC003"},
    }
    
    for _, tt := range tests {
        matched := false
        for _, rule := range rules {
            if rule.Pattern.MatchString(tt.code) {
                if rule.ID != tt.wantID {
                    t.Errorf("Rule %s matched for %q, want %s", rule.ID, tt.code, tt.wantID)
                }
                matched = true
                break
            }
        }
        if !matched {
            t.Errorf("No rule matched for: %q", tt.code)
        }
    }
}

package sandbox

import (
    "context"
    "os/exec"
    "time"
)

type WorkdirSandbox struct {
    cfg Config
}

func (s *WorkdirSandbox) Level() Level { return LevelWorkdir }

func (s *WorkdirSandbox) Run(ctx context.Context, cmd string) (*Result, error) {
    workDir := s.cfg.WorkDir
    if workDir == "" {
        workDir = os.TempDir()
    }

    timeout := 5 * time.Minute
    if s.cfg.Timeout > 0 {
        timeout = s.cfg.Timeout
    }
    ctx, cancel := context.WithTimeout(ctx, timeout)
    defer cancel()

    execCmd := exec.CommandContext(ctx, "sh", "-c", cmd)
    execCmd.Dir = workDir

    out, err := execCmd.CombinedOutput()
    result := &Result{
        Stdout:   string(out),
        ExitCode: 0,
    }
    if err != nil {
        result.ExitCode = -1
    }

    return result, nil
}

func (s *WorkdirSandbox) Cleanup() error { return nil }
```
